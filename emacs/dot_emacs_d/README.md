
# Config
    Put "init.el" into "~/.emacs.d/init.el"

# Install packages
    Add melpa repositories by running script "install_packages.el":
        C-x C-f ~/.emacs.d/install_packages_24.el
        M-x eval-buffer
            M-x -> Esc-x
            if space doesn't work use 'Esc-space'
    Restart Emacs
